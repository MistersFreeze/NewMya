local function getEquippedBulletSpeed()
    local sel = LocalPlayer.CurrentSelectedObject and LocalPlayer.CurrentSelectedObject.Value
    if typeof(sel) ~= "Instance" then return 2200 end

    local slot = LocalPlayer.GunInventory:FindFirstChild(sel.Name)
    if not slot or not slot.Value then return 2200 end

    local gunName
    if typeof(slot.Value) == "string" then
        gunName = slot.Value
    elseif typeof(slot.Value) == "Instance" then
        gunName = tostring(slot.Value)
    else
        return 2250
    end

    local data = ReplicatedStorage.GunData:FindFirstChild(gunName)
    if not data or not data:FindFirstChild("Stats") or not data.Stats:FindFirstChild("BulletSettings") then 
        return 2200 
    end

    local spd = data.Stats.BulletSettings:FindFirstChild("BulletSpeed")
    return (spd and spd:IsA("IntValue")) and spd.Value or 2200
end

local function calculateDrop(distance, speed)
    if not speed or speed <= 0 then return 0 end
    local t = distance / speed
    return 0.5 * aimbot.GRAVITY * t * t
end

-- ESP Functions
local function RemoveESP(player)
    if ESPs[player] then
        if ESPs[player].Connection then
            ESPs[player].Connection:Disconnect()
        end
        for _, drawing in pairs(ESPs[player].Drawings) do
            drawing.Visible = false
            drawing:Remove()
        end
        ESPs[player] = nil
    end
end

local function ESP(Player)
    RemoveESP(Player)

    local ESPConnection

    local Drawings = {
        Box = Drawing.new('Square'),
        BoxOutline = Drawing.new('Square'),
        Name = Drawing.new('Text'),
        Distance = Drawing.new('Text'),
    }

    Drawings.Name.Color = nameColor
    Drawings.Name.Center = true
    Drawings.Name.Outline = true
    Drawings.Name.OutlineColor = Color3.fromRGB(0, 0, 0)
    Drawings.Name.Size = 14
    Drawings.Name.Visible = false

    Drawings.Distance.Color = distanceColor
    Drawings.Distance.Center = true
    Drawings.Distance.Outline = true
    Drawings.Distance.OutlineColor = Color3.fromRGB(0, 0, 0)
    Drawings.Distance.Size = 14
    Drawings.Distance.Visible = false

    local function UpdateESP()
        ESPConnection = RunService.RenderStepped:Connect(function()
            if not ESPEnabled then
                for _, drawing in pairs(Drawings) do
                    drawing.Visible = false
                end
                return
            end

            local Character = Player.Character
            if Character then
                local HumanoidRootPart = Character:FindFirstChild('HumanoidRootPart')
                if HumanoidRootPart then
                    local distance = (HumanoidRootPart.Position - CurrentCamera.CFrame.Position).Magnitude

                    if distance > maxESPDistance then
                        Drawings.Box.Visible = false
                        Drawings.BoxOutline.Visible = false
                        Drawings.Name.Visible = false
                        Drawings.Distance.Visible = false
                        return
                    end

                    -- Ajuste para mover el ESP más arriba
                    local adjustedPosition = HumanoidRootPart.Position + Vector3.new(0, 1.5, 0)
                    local Position, OnScreen = CurrentCamera:WorldToViewportPoint(adjustedPosition)

                    if OnScreen then
                        local scale = 1 / (Position.Z * math.tan(math.rad(CurrentCamera.FieldOfView * 0.5)) * 2) * 1000
                        local width, height = math.floor(4.5 * scale), math.floor(6 * scale)
                        local x, y = math.floor(Position.X), math.floor(Position.Y)
                        -- Ajuste adicional para elevar el ESP
                        local yPosition = math.floor((y - height * 0.5) + (0.5 * scale) - 15)
                        local xPosition = math.floor(x - width * 0.5)

                        Drawings.Box.Size = Vector2.new(width, height)
                        Drawings.Box.Position = Vector2.new(xPosition, yPosition)
                        Drawings.Box.Visible = boxEnabled
                        Drawings.Box.Color = boxColor
                        Drawings.Box.Thickness = 1
                        Drawings.Box.Filled = boxFilledEnabled
                        Drawings.Box.Transparency = boxFilledEnabled and 0.5 or 0

                        Drawings.BoxOutline.Size = Vector2.new(width, height)
                        Drawings.BoxOutline.Position = Vector2.new(xPosition, yPosition)
                        Drawings.BoxOutline.Visible = boxOutlineEnabled
                        Drawings.BoxOutline.Color = boxOutlineColor
                        Drawings.BoxOutline.Thickness = 2

                        Drawings.BoxOutline.ZIndex = 1
                        Drawings.Box.ZIndex = 2

                        Drawings.Name.Position = Vector2.new(x, yPosition - 15)
                        Drawings.Name.Text = "(" .. Player.Name .. ")"
                        Drawings.Name.Visible = nameEnabled
                        Drawings.Name.Color = nameColor
                        Drawings.Name.ZIndex = 3

                        Drawings.Distance.Position = Vector2.new(x, yPosition + height + 5)
                        Drawings.Distance.Text = string.format("(%.0f meters)", distance)
                        Drawings.Distance.Visible = distanceEnabled
                        Drawings.Distance.Color = distanceColor
                        Drawings.Distance.ZIndex = 3
                    else
                        Drawings.Box.Visible = false
                        Drawings.BoxOutline.Visible = false
                        Drawings.Name.Visible = false
                        Drawings.Distance.Visible = false
                    end
                else
                    Drawings.Box.Visible = false
                    Drawings.BoxOutline.Visible = false
                    Drawings.Name.Visible = false
                    Drawings.Distance.Visible = false
                end
            else
                Drawings.Box.Visible = false
                Drawings.BoxOutline.Visible = false
                Drawings.Name.Visible = false
                Drawings.Distance.Visible = false
            end
        end)
    end

    coroutine.wrap(UpdateESP)()

    ESPs[Player] = {
        Connection = ESPConnection,
        Drawings = Drawings
    }

    local function OnCharacterAdded(character)
        local humanoid = character:WaitForChild("Humanoid")
        humanoid.Died:Connect(function()
            RemoveESP(Player)
        end)

        coroutine.wrap(function()
            task.wait(0.1)
            ESP(Player)
        end)()
    end

    if Player.Character then
        OnCharacterAdded(Player.Character)
    end
    Player.CharacterAdded:Connect(OnCharacterAdded)
end

local function applyOffset()
    local selected = LocalPlayer.CurrentSelectedObject.Value
    if not selected then return end

    local gunName = selected.Name
    local invItem = LocalPlayer:FindFirstChild("GunInventory"):FindFirstChild(gunName)
    if not (invItem and invItem.Value) then return end

    local dataFolder = ReplicatedStorage:FindFirstChild("GunData"):FindFirstChild(invItem.Value.Name)
    if not dataFolder then return end

    local offsetVal = dataFolder.Stats:FindFirstChild("Offset")
    if not (offsetVal and offsetVal:IsA("Vector3Value")) then return end

    if not originalOffsets[invItem.Value.Name] then
        originalOffsets[invItem.Value.Name] = offsetVal.Value
    end

    offsetVal.Value = Vector3.new(offsetX, offsetY, offsetZ)
end

local function restoreOffsets()
    for gunName, original in pairs(originalOffsets) do
        local dataFolder = ReplicatedStorage:FindFirstChild("GunData"):FindFirstChild(gunName)
        if dataFolder then
            local offsetVal = dataFolder.Stats:FindFirstChild("Offset")
            if offsetVal and offsetVal:IsA("Vector3Value") then
                offsetVal.Value = original
            end
        end
    end
    table.clear(originalOffsets)
end

local function getBarrel()
    local vm = CurrentCamera:FindFirstChild("ViewModel")
    if vm then
        local item = vm:FindFirstChild("Item")
        if item then
            return item:FindFirstChild("Barrel")
        end
    end

end

local function createLightningBeam(originCFrame, dir, length)
    local startP = Instance.new("Part")
    startP.Size = Vector3.new(0.1,0.1,0.1)
    startP.Anchored = true
    startP.CanCollide = false
    startP.Transparency = 1
    startP.CFrame = CFrame.new(originCFrame.Position)
    startP.Parent = workspace

    local endPos = originCFrame.Position + dir * length
    local endP = Instance.new("Part")
    endP.Size = Vector3.new(0.1,0.1,0.1)
    endP.Anchored = true
    endP.CanCollide = false
    endP.Transparency = 1
    endP.CFrame = CFrame.new(endPos)
    endP.Parent = workspace

    local a0 = Instance.new("Attachment", startP)
    local a1 = Instance.new("Attachment", endP)

    local beam = Instance.new("Beam")
    beam.Attachment0 = a0
    beam.Attachment1 = a1
    beam.FaceCamera = true
    beam.Width0 = 0.4
    beam.Width1 = 0.4
    beam.Color = ColorSequence.new(beamColor)
    beam.Transparency = NumberSequence.new(0.9)
    beam.TextureMode = Enum.TextureMode.Stretch
    beam.Segments = 20
    beam.Parent = workspace

    Debris:AddItem(startP, 1)
    Debris:AddItem(endP, 1)
    Debris:AddItem(beam, 1)
end

local function createTracer()
    local barrel = getBarrel()
    local originCFrame = barrel
        and (barrel.CFrame * CFrame.new(0,0, barrel.Size.Z/2))
        or CurrentCamera.CFrame

    local dir = CurrentCamera.CFrame.LookVector
    local length = 1000

    createLightningBeam(originCFrame, dir, length)
end

    if firing then return end

        while firing and tracerEnabled do
            createTracer()
            RunService.RenderStepped:Wait()
        end
    end)
end

end

local function updateGraveESP()
    local camera = Workspace.CurrentCamera
    
    -- Buscar tumbas nuevas
    for _, model in ipairs(Workspace:GetChildren()) do
        if model:IsA("Model") and model.Name == "Default" and not graveESP.ActiveDrawings[model] then
            local meshPart = model:FindFirstChildWhichIsA("MeshPart")
            if meshPart then
                createGraveESP(model, meshPart)
            end
        end
    end

    -- Actualizar tumbas existentes
    for model, data in pairs(graveESP.ActiveDrawings) do
        if not model:IsDescendantOf(Workspace) or not data.part then
            removeGraveESP(model)
        else
            local screenPos, onScreen = camera:WorldToViewportPoint(data.part.Position)
            if onScreen and graveESP.Enabled then
                local name = data.part:GetAttribute("DisplayName") or "Tumba desconocida"
                local searched = data.part:GetAttribute("Searched")
                local distanceStr = ""

                if typeof(searched) == "boolean" and searched == true then
                    name = "🔍 " .. name .. " (Revisada)"
                end

                if graveESP.ShowDistance then
                    local dist = (camera.CFrame.Position - data.part.Position).Magnitude
                    distanceStr = string.format(" [%.1fm]", dist)
                end

                data.text.Text = "⚰️ " .. name .. distanceStr
                data.text.Position = Vector2.new(screenPos.X, screenPos.Y)
                data.text.Size = graveESP.Size
                data.text.Color = graveESP.Color
                data.text.Outline = graveESP.Outline
                data.text.Visible = true
            else
                data.text.Visible = false
            end
        end
    end
end

    if type(str) ~= "string" then
        return tostring(str)
    end

end

local function scan_for_ammo_matches()
    if not ammoESP.Enabled then return end

    -- Acceso seguro a world_assets
    local worldAssets = Workspace:FindFirstChild("world_assets")
    if not worldAssets then return end
    
    local staticObjects = worldAssets:FindFirstChild("StaticObjects")
    if not staticObjects then return end
    
    local miscFolder = staticObjects:FindFirstChild("Misc")
    if not miscFolder then return end
    
    local ammoFolder = ReplicatedStorage:WaitForChild("Assets", 5):WaitForChild("ViewModels", 5):WaitForChild("Combat", 5):WaitForChild("Ammo", 5)

    for _, misc_child in ipairs(miscFolder:GetChildren()) do
        if misc_child:IsA("Model") then
            local source = misc_child:FindFirstChild("Main")
            if source and source:IsA("MeshPart") then
                local src_mesh_id = extract_id(source.MeshId)
                local src_tex_id = extract_id(source.TextureID)
                
                for _, ammo_model in ipairs(ammoFolder:GetChildren()) do
                    if ammo_model:IsA("Model") and ammo_model.Name ~= "Model" then
                        local ammo_main = ammo_model:FindFirstChild("Main")
                        if ammo_main and ammo_main:IsA("MeshPart") then
                            local ammo_mesh_id = extract_id(ammo_main.MeshId)
                            local ammo_tex_id = extract_id(ammo_main.TextureID)
                            
                            if ammo_mesh_id == src_mesh_id and ammo_tex_id == src_tex_id then
                                create_ammo_esp_on_source(source, ammo_model.Name)
                            end
                        end
                    end
                end
            end
        end
    end
end

local function getMeshPartsInfo(model)
    local info = {}
    for _, child in ipairs(model:GetChildren()) do
        if child:IsA("MeshPart") then
            local cNames = {}
            for _, c in ipairs(child:GetChildren()) do
                table.insert(cNames, c.Name)
            end
            table.sort(cNames)
            table.insert(info, { Name = child.Name, MeshId = child.MeshId, ChildrenNames = cNames })
        end
    end
    table.sort(info, function(a,b) return a.Name < b.Name end)
    return info
end

local function meshPartsMatch(a, b)
    if #a ~= #b then return false end
    for i = 1, #a do
        if a[i].Name ~= b[i].Name or a[i].MeshId ~= b[i].MeshId then
            return false
        end
        local c1, c2 = a[i].ChildrenNames, b[i].ChildrenNames
        if #c1 ~= #c2 then return false end
        for j = 1, #c1 do
            if c1[j] ~= c2[j] then return false end
        end
    end
    return true
end

local function getChildrenNames(model)
    local names = {}
    for _, c in ipairs(model:GetChildren()) do
        table.insert(names, c.Name)
    end
    table.sort(names)
    return names
end

local function childrenNamesMatch(a, b)
    if #a ~= #b then return false end
    for i = 1, #a do
        if a[i] ~= b[i] then return false end
    end
    return true
end

local function getDescendantsNames(model)
    local names = {}
    for _, d in ipairs(model:GetDescendants()) do
        table.insert(names, d.Name)
    end
    table.sort(names)
    return names
end

local function descendantsNamesMatch(a, b)
    if #a ~= #b then return false end
    for i = 1, #a do
        if a[i] ~= b[i] then return false end
    end
    return true
end

local function checkModelAgainstGunData(model)
    if not model:IsA("Model") then return nil end
    local meshInfo = getMeshPartsInfo(model)
    if #meshInfo > 0 then
        for name, refInfo in pairs(gunDataMeshInfos) do
            if meshPartsMatch(meshInfo, refInfo) then
                return name
            end
        end
    else
        local childNames = getChildrenNames(model)
        local candidates = {}
        for name, _ in pairs(gunDataMeshInfos) do
            local gd = gunDataFolder:FindFirstChild(name)
            if gd and gd:FindFirstChild("WorldModel") then
                if childrenNamesMatch(childNames, getChildrenNames(gd.WorldModel)) then
                    table.insert(candidates, name)
                end
            end
        end
        if #candidates == 1 then
            return candidates[1]
        elseif #candidates > 1 then
            local descNames = getDescendantsNames(model)
            for _, name in ipairs(candidates) do
                if descendantsNamesMatch(descNames, getDescendantsNames(gunDataFolder[name].WorldModel)) then
                    return name
                end
            end
        end
    end
    return nil
end

local function checkAndCreateWeaponESPForModel(model)
    local gunName = checkModelAgainstGunData(model)
    if gunName then
        if weaponsESP.ExcludeMelee and meleeBlacklist[gunName] then return end
        createWeaponESP(model, gunName)
    end
end

local function initialWeaponScan()
    -- Acceso seguro a world_assets
    local worldAssets = Workspace:FindFirstChild("world_assets")
    if not worldAssets then return end
    
    local staticObjects = worldAssets:FindFirstChild("StaticObjects")
    if not staticObjects then return end
    
    local miscFolder = staticObjects:FindFirstChild("Misc")
    if not miscFolder then return end
    
    for _, model in ipairs(miscFolder:GetChildren()) do
        checkAndCreateWeaponESPForModel(model)
    end
end

local function updateWeaponsESP()
    if not weaponsESP.Enabled then
        for model, data in pairs(weaponsESP.ActiveDrawings) do
            if data.drawing then
                data.drawing.Visible = false
            end
        end
        return
    end
    
    local camera = Workspace.CurrentCamera
    local camPos = camera.CFrame.Position
    
    for model, data in pairs(weaponsESP.ActiveDrawings) do
        local drawing = data.drawing
        local handle = model:FindFirstChild("Handle")
        
        if not handle or not model.Parent then
            drawing:Remove()
            weaponsESP.ActiveDrawings[model] = nil
        else
            local dist = (handle.Position - camPos).Magnitude
            local pos, onScreen = camera:WorldToViewportPoint(handle.Position)
            
            if onScreen and dist <= weaponsESP.RenderRange then
                if weaponsESP.ShowDistance then
                    drawing.Text = string.format("🔫 %s [%d m]", data.name, math.floor(dist))
                else
                    drawing.Text = "🔫 " .. data.name
                end
                drawing.Position = Vector2.new(pos.X, pos.Y)
                drawing.Visible = true
            else
                drawing.Visible = false
            end
        end
    end
end

-- Tracer system
local activeTracers = {}
local mouseButtonConnection = nil
local mouseHoldConnection = nil
local heartbeatConnection = nil
local lastFireTime = 0

local function createLightningTracer(startPos, endPos)
    local part = Instance.new("Part")
    part.Anchored = true
    part.CanCollide = false
    part.Transparency = 1
    part.Size = Vector3.new(0.1, 0.1, 0.1)
    part.CFrame = CFrame.new(startPos)
    part.Parent = Workspace

    local startAttach = Instance.new("Attachment")
    startAttach.Parent = part
    
    local endAttach = Instance.new("Attachment")
    endAttach.Position = endPos - startPos
    endAttach.Parent = part

    local beam = Instance.new("Beam")
    beam.Texture = "rbxassetid://446111271"
    beam.TextureSpeed = 3 + math.random() * 2
    beam.TextureLength = 0.5
    beam.LightEmission = 1
    beam.Color = ColorSequence.new(Options.LightningTracerColor.Value)
    beam.Width0 = Options.TracerThickness.Value
    beam.Width1 = Options.TracerThickness.Value * 0.8
    beam.FaceCamera = true
    beam.Transparency = NumberSequence.new(0)
    beam.Attachment0 = startAttach
    beam.Attachment1 = endAttach
    beam.Parent = part

    table.insert(activeTracers, {
        part = part,
        beam = beam,
        startTime = os.clock(),
        duration = Options.TracerDuration.Value
    })
end

local function updateTracers()
    local currentTime = os.clock()
    for i = #activeTracers, 1, -1 do
        local tracer = activeTracers[i]
        if currentTime - tracer.startTime >= tracer.duration then
            if tracer.part and tracer.part.Parent then
                tracer.part:Destroy()
            end
            table.remove(activeTracers, i)
        end
    end
end

local function fireTracer()
    local camera = Workspace.CurrentCamera
    if camera then
        createLightningTracer(
            camera.CFrame.Position,
            camera.CFrame.Position + (camera.CFrame.LookVector * 1000)
        )
    end
end

local function onMouseButton(input, gameProcessed)
    if not Toggles.LightningTracersToggle.Value or gameProcessed then
        return
    end
    
    if input.UserInputType == Enum.UserInputType.MouseButton1 then
        fireTracer()
        
        -- Setup automatic fire if enabled
        if Toggles.AutomaticFireToggle.Value then
            local fireDelay = 1 / Options.FireRate.Value
            mouseHoldConnection = RunService.Heartbeat:Connect(function()
                if os.clock() - lastFireTime >= fireDelay then
                    fireTracer()
                    lastFireTime = os.clock()
                end
            end)
        end
    end
end

local function onMouseButtonRelease(input, gameProcessed)
    if input.UserInputType == Enum.UserInputType.MouseButton1 and mouseHoldConnection then
        mouseHoldConnection:Disconnect()
        mouseHoldConnection = nil
    end
end

    if Toggles.LightningTracersToggle.Value then
        Library:Notify("Lightning Tracers Enabled", 3)
        mouseButtonConnection = UserInputService.InputBegan:Connect(onMouseButton)
        UserInputService.InputEnded:Connect(onMouseButtonRelease)

        end
        if mouseHoldConnection then
            mouseHoldConnection:Disconnect()
            mouseHoldConnection = nil
        end
        if heartbeatConnection then
            heartbeatConnection:Disconnect()
            heartbeatConnection = nil
        end
        
        -- Clean up active tracers
        for _, tracer in ipairs(activeTracers) do
            if tracer.part and tracer.part.Parent then
                tracer.part:Destroy()
            end
        end
        activeTracers = {}
    end
end)

local hitsoundsConnection
Toggles.HitsoundsToggle:OnChanged(function()
    local status = Toggles.HitsoundsToggle.Value
    if status then
        Library:Notify("Hitsounds On", 3)
        
        local ReplicatedStorage = game:GetService("ReplicatedStorage")
        local RunService = game:GetService("RunService")
        local Players = game:GetService("Players")
        local localPlayer = Players.LocalPlayer

        local SOUNDS_FOLDER_PATH = {
            "Assets",
            "Sounds",
            "DefaultHitmarker"
        }

        local SOUND_REPLACEMENTS = {
            Hitmarker = "rbxassetid://"..tostring(HITSOUND_OPTIONS[Options.HitmarkerSound.Value]),
            Headshot = "rbxassetid://"..tostring(HITSOUND_OPTIONS[Options.HeadshotSound.Value])
        }

        -- Sound replacement functions (original code)
        local function findSoundsFolder()
            local current = ReplicatedStorage
            for _, folderName in ipairs(SOUNDS_FOLDER_PATH) do
                current = current:FindFirstChild(folderName)
                if not current then return nil end
            end
            return current
        end

        local function replaceSounds()
            local soundsFolder
            while true do
                soundsFolder = findSoundsFolder()
                if soundsFolder then break end
                task.wait(1)
            end
            
            for soundName, newId in pairs(SOUND_REPLACEMENTS) do
                local sound = soundsFolder:FindFirstChild(soundName)
                if sound and sound:IsA("Sound") then
                    if not sound:FindFirstChild("__backup") then
                        local backup = sound:Clone()
                        backup.Name = "__backup"
                        backup.Parent = sound
                    end
                    sound.SoundId = newId
                    sound.Volume = 2.0
                end
            end

            while true do
                for soundName, newId in pairs(SOUND_REPLACEMENTS) do
                    local sound = soundsFolder:FindFirstChild(soundName)
                    if sound and sound:IsA("Sound") and sound.SoundId ~= newId then
                        sound.SoundId = newId
                    end
                end
                task.wait(0.5)
            end
        end

        local function interceptNewSounds()
            workspace.DescendantAdded:Connect(function(descendant)
                if descendant:IsA("Sound") then
                    task.wait(0.05)
                    for soundName, newId in pairs(SOUND_REPLACEMENTS) do
                        if descendant.Name == soundName then
                            descendant.SoundId = newId
                            descendant.Volume = 2.0
                        end
                    end
                end
            end)
        end

        -- Headshot notification system (works EXACTLY like hitsounds)
        local function setupHeadshotNotifications()
            local lastNotification = 0
            local headshotDebounce = {}
            
            local function onCharacterAdded(character)
                local humanoid = character:WaitForChild("Humanoid")
                
                humanoid:GetPropertyChangedSignal("Health"):Connect(function()
                    if humanoid.Health < humanoid.MaxHealth then
                        local hitPlayer = Players:GetPlayerFromCharacter(character)
                        if hitPlayer and hitPlayer ~= localPlayer then
                            if headshotDebounce[hitPlayer] and (tick() - headshotDebounce[hitPlayer] < 0.3) then
                                if tick() - lastNotification > 0.5 then
                                    lastNotification = tick()
                                    Library:Notify({
                                        Title = "💀 HEADSHOT",
                                        Content = hitPlayer.Name,
                                        Time = 3
                                    })
                                end
                            end
                        end
                    end
                end)
                
                if character:FindFirstChild("Head") then
                    character.Head.Touched:Connect(function()
                        local hitPlayer = Players:GetPlayerFromCharacter(character)
                        if hitPlayer then
                            headshotDebounce[hitPlayer] = tick()
                        end
                    end)
                end
            end
            
            Players.LocalPlayer.CharacterAdded:Connect(onCharacterAdded)
            if Players.LocalPlayer.Character then
                onCharacterAdded(Players.LocalPlayer.Character)
            end
        end

        -- Update sounds when dropdown changes
        Options.HitmarkerSound:OnChanged(function(value)
            SOUND_REPLACEMENTS.Hitmarker = "rbxassetid://"..tostring(HITSOUND_OPTIONS[value])
            if Toggles.HitsoundsToggle.Value then
                playSoundPreview(HITSOUND_OPTIONS[value])
            end
        end)

        Options.HeadshotSound:OnChanged(function(value)
            SOUND_REPLACEMENTS.Headshot = "rbxassetid://"..tostring(HITSOUND_OPTIONS[value])
            if Toggles.HitsoundsToggle.Value then
                playSoundPreview(HITSOUND_OPTIONS[value])
            end
        end)

        -- Start all systems together
        hitsoundsConnection = {
            replace = task.spawn(replaceSounds),
            intercept = task.spawn(interceptNewSounds),
            headshot = task.spawn(setupHeadshotNotifications) -- Added notification system
        }
    else
        Library:Notify("Hitsounds Off", 3)
        -- Clean up all systems
        if hitsoundsConnection then
            if hitsoundsConnection.replace then pcall(function() task.cancel(hitsoundsConnection.replace) end) end
            if hitsoundsConnection.intercept then pcall(function() hitsoundsConnection.intercept:Disconnect() end) end
            if hitsoundsConnection.headshot then pcall(function() hitsoundsConnection.headshot:Disconnect() end) end
            hitsoundsConnection = nil
        end
    end
end)

        if UserInputService:IsKeyDown(Enum.KeyCode.S) then moveVector -= cameraDirection end
        if UserInputService:IsKeyDown(Enum.KeyCode.A) then moveVector -= camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.D) then moveVector += camera.CFrame.RightVector end
        if UserInputService:IsKeyDown(Enum.KeyCode.Space) then moveVector += Vector3.new(0, 1, 0) end
        if UserInputService:IsKeyDown(Enum.KeyCode.LeftShift) then moveVector -= Vector3.new(0, 1, 0) end

        if moveVector.Magnitude > 0 then moveVector = moveVector.Unit end
        torso.CFrame += moveVector * Options.ManipSpeed.Value
        anchorPart.CFrame = torso.CFrame * CFrame.new(0, -10, 0)

        local distance = (torso.Position - startPosition).Magnitude
        if distance > 18 then
            distanceLabel.Text = ("YOU WILL HIT INVALIDS %.2f studs"):format(distance)
            distanceLabel.Color = Color3.new(1, 0, 0)
        elseif distance > 16 then
            distanceLabel.Text = ("You may get invalids %.2f studs"):format(distance)
            distanceLabel.Color = Color3.new(1, 1, 0)
        else
            distanceLabel.Text = ("Distance: %.2f studs"):format(distance)
            distanceLabel.Color = Color3.new(1, 1, 1)
        end
    end

-- Crosshair rendering
local rainbow = 0
local pulse = 0
local rotation = 0
local camera = workspace.CurrentCamera

game:GetService("RunService").RenderStepped:Connect(function(delta)
    -- Get position (screen center or mouse)
    local position = crosshair.ScreenCenter and 
        Vector2.new(camera.ViewportSize.X/2, camera.ViewportSize.Y/2) or
        Vector2.new(game:GetService("UserInputService"):GetMouseLocation().X, 
                  game:GetService("UserInputService"):GetMouseLocation().Y)

    if not crosshair.Enabled then
        for _, obj in pairs(drawings) do
            if type(obj) == "table" then
                for _, drawing in pairs(obj) do
                    drawing.Visible = false
                end
            else
                obj.Visible = false
            end
        end
        return
    end

    -- Rainbow effect
    rainbow = rainbow + delta * 0.5
    if rainbow > 1 then rainbow = 0 end
    local currentColor = crosshair.Rainbow and Color3.fromHSV(rainbow, 1, 1) or crosshair.Color

    -- Pulse effect
    if crosshair.Pulse then
        pulse = pulse + delta * crosshair.PulseSpeed
        if pulse > math.pi * 2 then pulse = 0 end
    end
    local pulseMod = crosshair.Pulse and (math.sin(pulse) * 0.25 + 0.75) or 1

    -- Rotation effect
    rotation = rotation + delta * crosshair.RotationSpeed
    if rotation > math.pi * 2 then rotation = 0 end

    -- Clear all drawings
    for _, obj in pairs(drawings) do
        if type(obj) == "table" then
            for _, drawing in pairs(obj) do
                drawing.Visible = false
            end
        else
            obj.Visible = false
        end
    end

    -- Draw based on style
    if crosshair.Style == "Default" then
        -- Default 4-line crosshair with effects
        for i = 1, 4 do
            local angle = rotation + math.rad(i * 90)
            local direction = Vector2.new(math.cos(angle), math.sin(angle))
            local startPos = position + (direction * crosshair.Gap)
            local endPos = position + (direction * crosshair.Size * pulseMod)
            
            drawings.lines[i].main.Visible = true
            drawings.lines[i].main.From = startPos
            drawings.lines[i].main.To = endPos
            drawings.lines[i].main.Color = currentColor
            drawings.lines[i].main.Thickness = crosshair.Thickness
            
            if crosshair.Outline then
                drawings.lines[i].outline.Visible = true
                drawings.lines[i].outline.From = startPos - Vector2.new(1, 1)
                drawings.lines[i].outline.To = endPos - Vector2.new(1, 1)
                drawings.lines[i].outline.Color = Color3.new(0, 0, 0)
                drawings.lines[i].outline.Thickness = crosshair.Thickness + 2
            end
        end
        
    elseif crosshair.Style == "Circle" then
        -- Circle style with effects
        drawings.circle.Visible = true
        drawings.circle.Position = position
        drawings.circle.Radius = crosshair.Size * pulseMod
        drawings.circle.Color = currentColor
        drawings.circle.Thickness = crosshair.Thickness
        
    elseif crosshair.Style == "Dot" then
        -- Dot style with pulse effect
        drawings.dot.Visible = true
        drawings.dot.Position = position
        drawings.dot.Color = currentColor
        drawings.dot.Radius = (crosshair.Size * 0.5) * pulseMod
        
    elseif crosshair.Style == "Swastika" then
        -- Swastika style with rotation
        for i = 1, 4 do
            local angle = rotation + math.rad(i * 90)
            local direction = Vector2.new(math.cos(angle), math.sin(angle))
            local startPos = position + (direction * crosshair.Gap)
            local midPos = position + (direction * crosshair.Size * 0.5 * pulseMod)
            local endPos = position + (direction * crosshair.Size * pulseMod)
            
            -- Main line
            drawings.lines[i].main.Visible = true
            drawings.lines[i].main.From = startPos
            drawings.lines[i].main.To = midPos
            drawings.lines[i].main.Color = currentColor
            drawings.lines[i].main.Thickness = crosshair.Thickness
            
            -- Secondary line (perpendicular)
            local perpAngle = angle + math.rad(90)
            local perpDirection = Vector2.new(math.cos(perpAngle), math.sin(perpAngle))
            drawings.lines[i].outline.Visible = true
            drawings.lines[i].outline.From = midPos
            drawings.lines[i].outline.To = midPos + (perpDirection * crosshair.Size * 0.5 * pulseMod)
            drawings.lines[i].outline.Color = currentColor
            drawings.lines[i].outline.Thickness = crosshair.Thickness
        end
    end

    -- Center dot (for non-dot styles)
    if crosshair.Dot and crosshair.Style ~= "Dot" then
        drawings.dot.Visible = true
        drawings.dot.Position = position
        drawings.dot.Color = currentColor
        drawings.dot.Radius = crosshair.Thickness * pulseMod
    end
end)

local veiculosData = {}
local NOME_MODELO_CARRO = "WorldModel"
local RENDER_DIST = 4000
local COR_NOME = Color3.fromRGB(255, 0, 0)
local COR_DIST = Color3.fromRGB(255, 105, 180)
local TEXTO_VEIC = "Car"
local HEIGHT_OFFSET = 5

local function atualizarVeiculos()
    -- Limpiar la tabla de vehículos
    for _, info in ipairs(veiculosData) do
        if info.txt then pcall(function() info.txt:Remove() end) end
        if info.dist then pcall(function() info.dist:Remove() end) end
    end
    table.clear(veiculosData)

    -- Buscar nuevos vehículos
    for _, obj in ipairs(workspace:GetChildren()) do
        if obj:IsA("Model") and obj.Name == NOME_MODELO_CARRO then
            table.insert(veiculosData, {modelo = obj, txt = nil, dist = nil})
        end
    end
end

    if c:IsA("Model") and c.Name == NOME_MODELO_CARRO then
        atualizarVeiculos()
    end
end)

    if c:IsA("Model") and c.Name == NOME_MODELO_CARRO then
        atualizarVeiculos()
    end
end)

atualizarVeiculos()

local frameSkip = 0
local function UpdateCarESP()
    if not carEspEnabled then
        for _, info in ipairs(veiculosData) do
            if info.txt then pcall(function() info.txt.Visible = false end) end
            if info.dist then pcall(function() info.dist.Visible = false end) end
        end
        return
    end

    frameSkip = (frameSkip + 1) % 2
    if frameSkip ~= 0 then return end

    for i = #veiculosData, 1, -1 do
        local info = veiculosData[i]
        if not info.modelo or not info.modelo.Parent then
            if info.txt then pcall(function() info.txt:Remove() end) end
            if info.dist then pcall(function() info.dist:Remove() end) end
            table.remove(veiculosData, i)
        end
    end

    for _, info in ipairs(veiculosData) do
        local ok, pivotCFrame = pcall(function() return info.modelo:GetPivot() end)
        if not ok or not pivotCFrame then
            if info.txt then info.txt.Visible = false end
            if info.dist then info.dist.Visible = false end
        else
            local pivotPos = pivotCFrame.Position
            local displayPos = pivotPos + Vector3.new(0, HEIGHT_OFFSET, 0)
            local dist = (CurrentCamera.CFrame.Position - pivotPos).Magnitude
            local pos2D, onscreen = CurrentCamera:WorldToViewportPoint(displayPos)

            if dist > RENDER_DIST or not onscreen or pos2D.Z <= 0 then
                if info.txt then info.txt.Visible = false end
                if info.dist then info.dist.Visible = false end
            else
                if not info.txt then
                    info.txt = Drawing.new("Text")
                    info.txt.Font = 2
                    info.txt.Size = 19
                    info.txt.Color = COR_NOME
                    info.txt.Center = true
                    info.txt.Outline = true
                end
                info.txt.Text = TEXTO_VEIC
                info.txt.Position = Vector2.new(pos2D.X, pos2D.Y)
                info.txt.Visible = true

                if not info.dist then
                    info.dist = Drawing.new("Text")
                    info.dist.Font = 1
                    info.dist.Size = 15
                    info.dist.Color = COR_DIST
                    info.dist.Center = true
                    info.dist.Outline = true
                end
                info.dist.Text = string.format("(%dm)", math.floor(dist/3.6))
                info.dist.Position = Vector2.new(pos2D.X, pos2D.Y + 15)
                info.dist.Visible = true
            end
        end
    end
end

Toggles.NoTreeToggle:OnChanged(function()
    local status = Toggles.NoTreeToggle.Value
    if status then
        Library:Notify("No Tree On", 3)
        
        -- Start the No Tree script
        local lastCheck = 0
        local cache = {}

        noTreeConnection = RunService.Heartbeat:Connect(function()
            local currentTime = tick()

            if currentTime - lastCheck > 5 or #cache == 0 then
                lastCheck = currentTime
                cache = {}

                for _, part in ipairs(Workspace:GetDescendants()) do
                    if part:IsA("BasePart") and (part.Name == "Leaves" or part.Name == "Tree") then
                        table.insert(cache, part)
                    end
                end

                for _, part in ipairs(cache) do
                    if part.Transparency ~= 1 then
                        part.Transparency = 1
                    end
                    if part.CanCollide ~= false then
                        part.CanCollide = false
                    end
                    if part.Anchored ~= true then
                        part.Anchored = true
                    end
                end
            end
        end)
    else
        Library:Notify("No Tree Off", 3)
        if noTreeConnection then
            noTreeConnection:Disconnect()
            noTreeConnection = nil
        end
        
        -- Restaurar los árboles
        for _, part in ipairs(Workspace:GetDescendants()) do
            if part:IsA("BasePart") and (part.Name == "Leaves" or part.Name == "Tree") then
                part.Transparency = 0
                part.CanCollide = true
                -- No modificamos Anchored porque no lo cambiamos en la activación.
            end
        end
    end
end)

-- Body Chams Functions
function aWIDJhawoidjawoidjawoidjawoidIsArm(part)
    local armParts = {"LeftUpperArm", "LeftLowerArm", "LeftHand", "RightUpperArm", "RightLowerArm", "RightHand"}
    return table.find(armParts, part.Name)
end

function aWIDJhawoidjawoidjawoidjawoidChamPart(part)
    if part:IsA('BasePart') then
        for _, child in ipairs(part:GetChildren()) do
            if child:IsA('SurfaceAppearance') then
                child:Destroy()
            end
        end
        part.Material = aWIDJhawoidjawoidjawoidjawoidChamMaterial
        part.Color = aWIDJhawoidjawoidjawoidjawoidIsArm(part) and aWIDJhawoidjawoidjawoidjawoidArmColor or aWIDJhawoidjawoidjawoidjawoidGunColor
    end
end

function aWIDJhawoidjawoidjawoidjawoidProcessModel(model)
    for _, child in ipairs(model:GetDescendants()) do
        if child:IsA('BasePart') then
            aWIDJhawoidjawoidjawoidjawoidChamPart(child)
        end
    end
end

function aWIDJhawoidjawoidjawoidjawoidChangeSkinLook()
    if not aWIDJhawoidjawoidjawoidjawoidChamsEnabled then return end
    
    local cam = workspace.Camera
    for _, obj in pairs(cam:GetDescendants()) do
        if obj.Name == "Skin" and obj:IsA("BasePart") then
            obj.Material = aWIDJhawoidjawoidjawoidjawoidChamMaterial
            obj.Color = aWIDJhawoidjawoidjawoidjawoidSkeletonColor
            
            for _, child in pairs(obj:GetChildren()) do
                if child:IsA("SurfaceAppearance") then
                    child:Destroy()
                end
            end
        end
    end
end

function aWIDJhawoidjawoidjawoidjawoidApplyChams()
    if not aWIDJhawoidjawoidjawoidjawoidChamsEnabled then return end

    local folder = workspace:FindFirstChild("Characters")
    if not folder then return end

    for _, character in pairs(folder:GetChildren()) do
        for _, child in ipairs(character:GetDescendants()) do
            if (child:IsA('Model') or child:IsA('WorldModel')) and child:IsDescendantOf(character) then
                aWIDJhawoidjawoidjawoidjawoidProcessModel(child)
            end
        end
        
        for _, part in ipairs(character:GetDescendants()) do
            if part:IsA('BasePart') then
                aWIDJhawoidjawoidjawoidjawoidChamPart(part)
            end
        end
    end
    
    aWIDJhawoidjawoidjawoidjawoidChangeSkinLook()
end

    if not aWIDJhawoidjawoidjawoidjawoidWeaponChamsEnabled then return end

            v.Color = aWIDJhawoidjawoidjawoidjawoidWeaponChamsColor
        end

        if v:IsA("SurfaceAppearance") then
            v:Destroy()
        end
    end
end

                    drawing_text.Text = "📦 " .. data.label
                end

                local screen_pos, on_screen = camera:WorldToViewportPoint(source_part.Position)
                if on_screen then
                    drawing_text.Position = Vector2.new(screen_pos.X, screen_pos.Y)
                    drawing_text.Visible = true
                else
                    drawing_text.Visible = false
                end
            end
        end
    end
end)

-- Main RenderStepped loop para el nuevo aimbot
RunService.RenderStepped:Connect(function()
    -- Actualizar círculo FOV
    local t = tick()
    local center2d = Vector2.new(CurrentCamera.ViewportSize.X / 2, CurrentCamera.ViewportSize.Y / 2)
    fovCircle.Position = center2d
    fovCircle.Visible = aimbot.showFOV
    fovCircle.Color = aimbot.fovRainbow and Color3.fromHSV(t % 5 / 5, 1, 1) or aimbot.fovColor
    
    -- Obtener cabezas de jugadores
    local heads = {}
    for _, mdl in ipairs(Workspace.Characters:GetChildren()) do
        if mdl:IsA("Model") and mdl:FindFirstChild("Head") then
            table.insert(heads, mdl.Head)
        end
    end
    
    -- Excluir cabeza del jugador local
    local myHead = LocalPlayer.Character and (LocalPlayer.Character:FindFirstChild("ServerColliderHead") or LocalPlayer.Character:FindFirstChild("Head"))
    local ignoreIdx, ignoreDist = nil, math.huge
    if myHead then
        for i, head in ipairs(heads) do
            local d = (head.Position - myHead.Position).Magnitude
            if d < ignoreDist then
                ignoreDist = d
                ignoreIdx = i
            end
        end
    end
    
    -- Encontrar objetivo más cercano dentro del FOV
    local targetHead, target2d, minDist = nil, nil, math.huge
    for i, head in ipairs(heads) do
        if i ~= ignoreIdx then
            local pos2d, onScreen = worldToScreen(head.Position)
            if onScreen then
                local dist = (pos2d - center2d).Magnitude
                if dist <= aimbot.fovRadius and dist < minDist then
                    minDist = dist
                    targetHead = head
                    target2d = pos2d
                end
            end
        end
    end
    
    -- Lógica de apuntado
    if aimbot.aiming and aimbot.enabled and targetHead then
        local camPos = CurrentCamera.CFrame.Position
        local headPos = targetHead.Position
        local horizDist = (Vector3.new(headPos.X, camPos.Y, headPos.Z) - camPos).Magnitude
        local speed = getEquippedBulletSpeed()
        local drop = calculateDrop(horizDist, speed) * 1.4 * aimbot.predictionYAdjust
        
        -- Calcular predicción de movimiento
        local predictedOffset = Vector3.zero
        for _, plr in ipairs(Players:GetPlayers()) do
            if plr ~= LocalPlayer and plr.Character and plr.Character:FindFirstChild("ServerColliderHead") then
                local velocity = plr.Character.ServerColliderHead.Velocity
                local predictedTime = horizDist / speed
                local prediction = velocity * predictedTime * aimbot.predictionXAdjust
                if (plr.Character.ServerColliderHead.Position - headPos).Magnitude < 10 then
                    predictedOffset = prediction
                    break
                end
            end
        end
        
        -- Calcular posición compensada
        local compensatedPos = headPos + Vector3.new(0, drop * 1.5, 0) + predictedOffset
        local comp2d, onS = worldToScreen(compensatedPos)
        
        -- Mover mouse hacia el objetivo
        if onS then
            local delta = comp2d - center2d
            if mousemoverel then
                mousemoverel(delta.X * aimbot.smoothing, delta.Y * aimbot.smoothing)
            end
        end
    end
    
    -- Fullbright
    if fullbrightEnabled then
        lighting.Brightness = 10
        lighting.ClockTime = 12
        lighting.GlobalShadows = false
        lighting.OutdoorAmbient = Color3.fromRGB(255, 255, 255)
    end

    -- No Fog
    if noFogEnabled then
        if lighting:FindFirstChild("Atmosphere") then
            lighting.Atmosphere.Density = 0
        end
    end

    -- Time Changer
    if timeEnabled then
        lighting.ClockTime = timeValue
    end
end)

    end

        workspace.Terrain.Clouds.Enabled = originalCloudsEnabled
    end

        resetFov()
